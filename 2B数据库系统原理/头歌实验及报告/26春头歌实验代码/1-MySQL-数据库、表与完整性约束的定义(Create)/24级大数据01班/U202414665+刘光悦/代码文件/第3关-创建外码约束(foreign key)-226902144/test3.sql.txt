CREATE DATABASE IF NOT EXISTS MyDb;
USE MyDb;

-- 2. 创建部门表
CREATE TABLE dept (
    deptNo   INT           NOT NULL,
    deptName VARCHAR(32),
    PRIMARY KEY (deptNo)
);

-- 3. 创建职工表，并添加外键约束 FK_staff_deptNo
CREATE TABLE staff (
    staffNo   INT             NOT NULL,
    staffName VARCHAR(32),
    gender    CHAR(1),
    dob       DATE,
    salary    DECIMAL(8,2),
    deptNo    INT,
    PRIMARY KEY (staffNo),
    CONSTRAINT FK_staff_deptNo
      FOREIGN KEY (deptNo)
      REFERENCES dept(deptNo)
);
